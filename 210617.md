# 아래도 정답인데, n+= 1을 왜 하는걸까?

m, n = map(int, input().split())


def prime_list(m, n):
    n += 1
    sieve = [True] * n
    for i in range(2, int(n**0.5)+1):
        if sieve[i] == True:
            for j in range(i+i, n, i):
                sieve[j] = False

    for i in range(m, n):
        if i > 1 and sieve[i] == True:
            print(i)


prime_list(m, n)
